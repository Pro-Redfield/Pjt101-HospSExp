SELECT
	tb_dependente.iddependente,
	tb_dependente.nome,
	tb_dependente.parentesco,
    tb_funcionario.idfuncionario,
    tb_funcionario.nome
FROM
   tb_dependente
INNER JOIN
   tb_funcionario ON tb_dependente.funcionario_id = tb_funcionario.idfuncionario
WHERE
   tb_dependente.iddependente = 3
